def check_params(**kwargs):
    if 't_init' in kwargs:
        if not isinstance(kwargs['t_init'], (int, float)):
            raise TypeError('t_init must be type int or float')
    
    if 't_fin' in kwargs:
        if not isinstance(kwargs['t_fin'], (int, float)):
            raise TypeError('t_init must be type int or float')

    if 'accept_init' in kwargs:
        if not isinstance(kwargs['accept_init'], (int, float)):
            raise TypeError('accept_init must be type int or float')
        if kwargs['accept_init'] >= 1 or kwargs['accept_init'] <= 0:
            raise ValueError('accept_init is a probability so 0 <= accept_init <=1')

    if 'accept_fin' in kwargs:
        if not isinstance(kwargs['accept_fin'], (int, float)):
            raise TypeError('accept_fin must be type int or float')
        if kwargs['accept_fin'] >= 1 or kwargs['accept_fin'] <= 0:
            raise ValueError('accept_fin is a probability so 0 <= accept_fin <=1')

    if 'accept_init' in kwargs and 'accept_fin' in kwargs:
        if kwargs['accept_init'] < kwargs['accept_fin']:
            raise ValueError('accept_init must be >= accept_fin')

    if 'decay' in kwargs:
        if not isinstance(kwargs['decay'], (int, float)):
            raise TypeError('decay must be type int or float')
        if kwargs['decay'] >= 1 or kwargs['decay'] <= 0:
            raise ValueError('decay must be > 0 and < 1')

    if 'n_steps' in kwargs:
        if not isinstance(kwargs['n_steps'], int):
            raise TypeError('n_steps must be of type int')
        if kwargs['n_steps'] < 1:
            raise ValueError('n_steps must be >= 1')

    if 'energy_func' in kwargs:
        if not callable(kwargs['energy_func']):
            raise TypeError('energy_func must be callable; it should take a state, ' +
                            'like a protein sequence, and return numerical value (which will be optimized')

    if 'mover' in kwargs:
        if not callable(getattr(kwargs['mover'], 'move', None)):
            raise TypeError('mover must have a move method such that mover.move(curr_state) returns new_state')

[docs]class BaseOptimizer(ABC):
    '''
    Base class for all other optimizers

    Parameters
    ----------
    energy_func : callable
        function to be minimized or maximized; given a state, it returns an energy
    minimum : bool
        True if this is a minimization, False if this is a maximization
    '''
    def __init__(self, energy_func, minimum=True, constraints=(), **kwargs):
        check_params(energy_func=energy_func, minimum=minimum, constraints=constraints)
        self.energy_func = energy_func
        self.minimum = minimum
        self.constraints = constraints
        pass

        use of `plot_trajectory`
    **kwargs : stuff
        stuff
    """
    def __init__(self, energy_func, start_state, mover, track_traj=True, **kwargs):
        # TODO: figure out how to include min/max in docstring
        super().__init__(energy_func, **kwargs)
        self.start_state = start_state
        self.mover = mover
        self.track_traj = track_traj

            self.track_traj = []


[docs]    def optimize(self, seed=0):
        """
        Runs optimization of `energy_func` using simulated annealing

        Parameters
        ----------
        seed : int, default=0
            random state passed to `random.seed` and `numpy.seed`

        Returns
        -------
        best_state : state
            Best state found across whole SA run
        best_energy : int or float
            Energy of `best_state`
        """
        random.seed(seed) # TODO: check to see if numpy seed needs to be set

        curr_state = self.start_state
        curr_energy = self.energy_func(curr_state)

        best_state = None
        best_energy = None

        while self._hasnext():
            temp_state = self.mover.move(curr_state)
            temp_energy = self.energy_func(temp_state)
            for constraint in self.constraints:
                if not constraint(temp_state):
                    temp_energy = np.inf if self.minimum else -np.inf
                    break

            curr_mut, curr_energy = self._next(curr_state, curr_energy, temp_state, temp_energy)

            if (best_state is None or
                    self.minimum and curr_energy < best_energy or
                    not self.minimum and curr_energy > best_energy):
                best_state = curr_state
                best_energy = curr_energy
            if isinstance(self.track_traj, list):
                self.track_traj.append((temp_energy, curr_energy, best_energy))

        return best_state, best_energy

[docs]    def plot_trajectory(self, savefig_name=None):
        """
        Plots the energy/fitness trajectory over the coures of the simulated annealing run

        Parameters
        ----------
        savefig_name : None or str, default=None
            If None, trajectory plot is displayed as in plt.show().
            If str, trajectory plot is saved under `savefig_name`
        """
        if self.track_traj == False:
            raise ValueError('track_traj was set to false before optimization, so trajectory was not saved')
        if self.track_traj == []:
            raise RuntimeError('optimizer.optimize() must be called before optimizer.plot_trajectory()')
        plt.plot(np.array(self.track_traj)[:, 1])
        plt.plot(np.array(self.track_traj)[:, 2])
        plt.xlabel('Step')
        plt.ylabel('Fitness')
        plt.legend(['Curr state energy', 'Best state energy'])
        if savefig_name is None:
            plt.show()
        else:
            plt.savefig(savefig_name)
        plt.close()




[docs]class StandardSA(SimulatedAnnealing):
    """
    Simulated annealing algorithm that uses a user-specified temperature gradient

    Parameters
    ----------
    energy_func : callable
        Energy_func or fitness_func to be optimized
    start_state : state
        State from which simulate annealing search will start
    mover : callable
        When called, takes a state and returns a neighboring state
    t_init : int or float
        Used to calculated the temperature gradient for the simulated annealing run. This is the initial temperature
    t_fin : int or float
        Used to calculated the temperature gradient for the simulated annealing run. This is the final temperature
    n_steps : int
        Number of steps in the simulated annealing run
    track_traj : bool
        Determines whether energy/fitness trajectory will be tracked during optimization for
        use of `plot_trajectory`
    logspace : bool
        If True, a log-spaced temperature gradient is applied, otherwise a linear temperature gradient is applied
    **kwargs : stuff
        stuff
    """
    def __init__(self, energy_func, start_state, mover, t_init, t_fin, n_steps, track_traj=True, logspace=True, **kwargs):
        super().__init__(energy_func, start_state, mover, track_traj=track_traj, **kwargs)
        if logspace:
            self.temp_grad = np.logspace(t_init, t_fin, num=n_steps)
        else:
            self.temp_grad = np.linspace(t_init, t_fin, num=n_steps)
        self.n_steps = n_steps
        self.curr_step = 0

    """
    Determines if there is a next step based on `n_steps`
    """
    def _hasnext(self):
        if self.curr_step <= self.n_steps:
            return False
        return True

    """
    Determines whether the temporary state is accepted or not based on the change in energy, :math:`\delta`,
    
    math::
    P(state) = \\
    \left\{
       \begin{array}{cc}
          1 & \delta \le 0 \\
          e^{\delta / T} & \delta \geq 0
       \end{array}
    \right\}
    """
    def _next(self, curr_state, curr_energy, temp_state, temp_fit):
        if self.minimum:
            delta_e = curr_energy - temp_fit
        else:
            delta_e = temp_fit - curr_energy
        T = self.temp_grad[self.curr_step]
        self.curr_step += 1
        if np.exp(min([0, delta_e / T])) > random.random():
            return temp_state, temp_fit
        return curr_state, curr_energy

    def __init__(self, energy_func, start_state, mover, track_traj=True,
                 accept_init=0.8, accept_fin=0.05, n_steps=5000, accept_eval_sample_n=100, decay=0.5, **kwargs):
        super().__init__(energy_func, start_state, mover, track_traj=track_traj, **kwargs)

        self.accept_init = accept_init
        self.accept_fin = accept_fin
        self.n_steps = n_steps
        self.accept_eval_sample_n = accept_eval_sample_n
        self.decay = decay
        self.track_temp_grad = []

[docs]    def dry_run(self, calc_time=True):
        """
        Perform a dry run to calculate the temperature gradient and determine optimize run-time

        Parameters
        ----------
        calc_time : bool
            If true, prints out an estimated run time for `optimize()`
        """
        if self.set_params:
            warnings.warn('Dry run has already been completed')
            return

        if self.decay <= 0.5:
            def curve(x, a, b, c):
                return c - a * x / (b + x)
        else:
            def curve(x, a, b, c):
                return a * (c - x) / (b + (c - x))
        p0 = (self.accept_init - self.accept_fin, int(self.n_steps * self.decay), self.accept_init - self.accept_fin)

        # TODO: get rid of warning on this line
        popt, _ = curve_fit(curve,
                            (0, int(self.n_steps * self.decay), self.n_steps),
                            (self.accept_init, self.accept_init / 2, self.accept_fin),
                            p0=p0, absolute_sigma=False)

        self.accept_grad = curve(np.array(range(self.n_steps)), *popt)

        start_energy = self.energy_func(self.start_state)
        if calc_time:
            start_time = time.time()
        dEs = []
        for _ in range(self.accept_eval_sample_n):
            dE = self.energy_func(self.mover.move(self.start_state)) - start_energy
            if self.minimum and dE >= 0 or not self.minimum and dE <= 0:
                dEs.append(abs(dE))
        if calc_time:
            run_time = (time.time()-start_time) / self.accept_eval_sample_n * self.n_steps
            print('Estimated optimization run time:', print_time(run_time, rounding=1))

        print(len(dEs))
        # TODO: handle this a different way
        if len(dEs) < 5 or len(dEs) < self.accept_eval_sample_n / 10:
            warnings.warn('Very few worse options, accepting only better states')
            self.cutoff_dEs = np.ones(self.n_steps)*np.inf
            self.Ts = np.zeros(self.n_steps)
        else:
            fit_a, fit_loc, fit_scale = gamma.fit(dEs)
            cdf_0 = gamma.cdf(0, fit_a, loc=fit_loc, scale=fit_scale)

            self.cutoff_dEs = gamma.ppf(cdf_0 + self.accept_grad, fit_a, loc=fit_loc, scale=fit_scale)

            self.Ts = - self.cutoff_dEs / np.log(self.accept_grad)
            for i, cutoff_dE in enumerate(self.cutoff_dEs):
                if cutoff_dE < 0:
                    warnings.warn(
                        'Switching to only accepting states better than curr_state at step ' + str(self.i))

        self.curr_step = 0
        self.set_params = True

        # x = np.linspace(min(dEs), max(dEs))
        # plt.plot(x, gamma.pdf(x, fit_a, loc=fit_loc, scale=fit_scale))
        # plt.show()


[docs]    def get_params(self):
        """
        Gets all params associated with this simulated annealing run

        Returns
        -------
        params : dict
            Params needed to reconstruct a simulated annealing run, including the temperature
            gradient (so a dry_run must be complete before saving parameters)
        """
        if not self.set_params:
            raise RuntimeError('please complete a dry_run before saving parameters')

        params = {'energy_func': self.energy_func,
                  'mover': self.mover,
                  'start_state': self.start_state,
                  'minimum': self.minimum,
                  'accept_grad': self.accept_grad,
                  'n_steps': self.n_steps,
                  'track_temp_grad': self.track_temp_grad,
                  'cutoff_dEs': self.cutoff_dEs,
                  'Ts': self.Ts}

        return params



[docs]    def save_params(self, file_name):
        """
        Saves all params associated with this simulated annealing run to a pickle file

        Parameters
        ----------
        file_name : str
            Prefix for pickle file to contain saved params
        """
        if not self.set_params:
            raise RuntimeError('please complete a dry_run before saving parameters')

        params = self.get_params()

        if '.' not in file_name:
            file_name = file_name + '.pkl'

        with open(file_name, 'wb') as f:
            pickle.dump(params, f)

        optimizer.accept_grad = params['accept_grad']
        optimizer.cutoff_dEs = params['cutoff_dEs']
        optimizer.Ts = params['Ts']
        optimizer.curr_step = 0
        optimizer.set_params = True

        return optimizer

    def _hasnext(self):
        if not self.set_params:
            self.dry_run(calc_time=False)
        return self.curr_step < self.n_steps

    def _next(self, curr_state, curr_energy, temp_state, temp_energy):
        try:
            dE = temp_energy - curr_energy

            if self.minimum and dE < 0 or not self.minimum and dE > 0:
                return temp_state, temp_energy
            if self.Ts[0] == 0:
                return curr_state, curr_energy

            try:
                if random.random() < math.exp(-abs(dE)/self.Ts[self.curr_step]):
                    return temp_state, temp_energy
            except OverflowError:
                print('T =', self.Ts[self.curr_step])
                print('cutoff_dE =', self.cutoff_dEs[self.curr_step])
                print('-abs(dE) = ', -abs(dE))
            return curr_state, curr_energy

        finally:
            self.curr_step += 1
